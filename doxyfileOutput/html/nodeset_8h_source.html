88 void AttemptMove(double rx, double ry) {
89
90 done = true;
91 // 1. Collect list of all attached to the nodes in the set
92 list<Cell *> celllist = getCells();
93
94 // 2. Sum the current energy of these cells
95 double old_energy=0.;
96 for ( list<Cell *>::const_iterator i = celllist.begin();
97 i!=celllist.end();
98 ++i ) {
99
100 old_energy += (*i)->Energy();
101 }
102
103 // 3. (Temporarily) move the set's nodes.
104 for ( list<Node *>::iterator n = begin();
105 n!=end();
106 ++n ) {
107
108 (*n)->x+=rx;
109 (*n)->y+=ry;
110 // (*n)->z += rz;
111 }
112
113 // 4. Recalculate the energy
114 double new_energy = 0.;
115 for ( list<Cell *>::const_iterator i = celllist.begin();
116 i!=celllist.end();
117 ++i ) {
118
119 new_energy += (*i)->Energy();
120 }
121
122
123 // 5. Accept or Reject DeltaH
124 double dh = new_energy - old_energy;
125
126 list<int> new_areas;
127
128 // cerr << "Nodeset says: dh = " << dh << " ...";
129 if (RANDOM()<exp((-dh)/par.T) ) {
130
131 // ACCEPT
132 // recalculate areas of cells
133 for_each ( celllist.begin(), celllist.end(), mem_fn ( &Cell::RecalcArea ) );
134
135 // recalculate integrals of cells
136 for_each ( celllist.begin(), celllist.end(), mem_fn ( &Cell::SetIntegrals ) );
137
138
139 } else {
140 // REJECT
141
142 // Move the set's nodes back to their original position
143 for ( list<Node *>::iterator n = begin();
144 n!= end();
145 ++n ) {
146
147 (*n)->x-=rx;
148 (*n)->y-=ry;
149 }
150 }
151 }
