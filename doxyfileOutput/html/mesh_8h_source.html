61 public:
62 typedef typename C::value_type value_type;

65 if (find (queue<T,C>::c.begin(),queue<T,C>::c.end(),x)==queue<T,C>::c.end()) {
66 queue<T,C>::c.push_back(x);
67 }
68 }

70 queue<T,C>::c.clear();

222 void DoCellHouseKeeping(vector<CellWallCurve>& curves) {
223 vector<Cell *> current_cells = cells;
224 for (vector<Cell *>::iterator i = current_cells.begin();
225 i != current_cells.end();
226 i ++) {
227 plugin->CellHouseKeeping(*i);
228 }
229 sort(curves.begin(), curves.end(), [](CellWallCurve lhs, CellWallCurve rhs) {return lhs.getThreshold() > rhs.getThreshold();});
230 CellWallCurve * array = &(curves[0]);
231 double count = curves.size();
232 for (int index = 0;index < (count*3); index++) {
233 double rand= RANDOM();
234 // quadratic random is used to prioritize high energy cases over low energy cases
235 int arrayIndex = round((count-1.)*rand*rand);
236 CellWallCurve * element= &(array[arrayIndex]);
237 if (element->removeSpike()) {
238 if(element->isBorderCase()){
239 RepairBoundaryPolygon();
240 }
241 bool anyActive = false;
242 for (std::vector<CellWallCurve>::iterator it2 = curves.begin(); it2 != curves.end(); ++it2){
243 it2->check_overlap(*element);
244 anyActive = anyActive || !(it2->isDeacivated());
245 }
246 element->reset();
247 if (!anyActive) {
248 break;
249 }
250 }
251 }
252
253 for (vector<Cell *>::iterator i = current_cells.begin();
254 i != current_cells.end();
255 i ++) {
256 // Call functions of Cell that cannot be called from CellBase, including Division
257 if ((*i)->flag_for_divide) {
258 if ((*i)->division_axis) {
259 (*i)->DivideOverAxis(*(*i)->division_axis);
260 delete (*i)->division_axis;
261 (*i)->division_axis = 0;
262 } else {
263 (*i)->Divide();
264 }
265 (*i)->flag_for_divide=false;
266 }
267 }

299 void IncreaseCellCapacityIfNecessary(void) {
300
301 return;
302 // cerr << "Entering Mesh::IncreaseCellCapacityIfNecessary \n";
303 // make sure we always have enough space
304 // to have each cell divide at least once
305 //
306 // Note that we must do this, because Cell::Divide pushes a new Cell
307 // onto Mesh::cells. As a result, Mesh::cells might be relocated
308 // if we are _within_ a Cell object: i.e. pointer "this" will be changed!!
309 //
310 // An alternative solution could be to make "Mesh::cells" a list,
311 // but this won't work because we need random access for
312 // the Monte Carlo algorithm.
313
314 if (2*cells.size()>cells.capacity()) {
315 cerr << "Increasing capacity to " << 2*cells.capacity() << endl;
316 cerr << "Current capacity is " << cells.capacity() << endl;
317 cells.reserve(cells.capacity()*2);
318 }
319 }
