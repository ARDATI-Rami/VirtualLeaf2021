259 template<class Op> void LoopWallElements(Op f) {
260 WallElementInfo * info = newWallElementInfo();
261 list <Node *>::iterator i=nodes.begin();
262 Node * first=*i;
263 Node * from=first;
264 Node * to=*(++i);
265 while (i!=nodes.end()) {
266 fillWallElementInfo(info,from,to);
267 f(info);
268 if (stopWallElementInfo(info)) {
269 delete info;
270 return;
271 }
272 from=to;
273 to=*(++i);
274 }
275 fillWallElementInfo(info,from,first);
276 f(info);
277 deleteWallElementInfo(info);
278 }

280 template<class Op> void LoopWallElementsOfWall(Wall* wall, Op f) {
281 WallElementInfo * info = newWallElementInfo();
282 list <Node *>::iterator i=nodes.begin();
283 Node * first=*i;
284 Node * from=first;
285 Node * to=*(++i);
286 bool start = false;
287 while (i!=nodes.end()) {
288 if (wall->isHasStartOrEnd(from)) {
289 if (start) {
290 deleteWallElementInfo(info);
291 return;
292 } else {
293 start = true;
294 }
295 }
296 if (start) {
297 fillWallElementInfo(info,from,to);
298 f(info);
299 if (stopWallElementInfo(info)) {
300 deleteWallElementInfo(info);
301 return;
302 }
303 }
304 from=to;
305 to=*(++i);
306 }
307 fillWallElementInfo(info,from,first);
308 f(info);
309 deleteWallElementInfo(info);
310 }
