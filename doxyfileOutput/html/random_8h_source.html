41// Included Random shuffle to retain compatibility with the MacOS standard library
42template <class RandomAccessIterator, class RandomNumberGenerator>
43 void vl_shuffle (RandomAccessIterator first, RandomAccessIterator last,
44 RandomNumberGenerator& gen) {
45 long i, n;
46 n = (last-first);
47 for (i=n-1; i>0; --i) {
48 swap (first[i],first[gen(i+1)]);
49 }
50}
