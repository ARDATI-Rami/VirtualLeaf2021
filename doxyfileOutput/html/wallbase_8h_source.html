40template<class In, class Out> Out duplicates_copy(In first, In last, Out res) {
41
42 In i = adjacent_find(first, last);
43
44 while (i!=last) {
45 res++ = *i;
46 i = adjacent_find(++i, last);
47 }

49 return res;
