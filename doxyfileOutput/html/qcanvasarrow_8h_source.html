40 void paint(QPainter *painter, const QStyleOptionGraphicsItem *option, QWidget *widget = nullptr){
41 // construct arrow head
42 QPointF start=line().p1();
43 QPointF end=line().p2();
44 QPointF mid=start + (3./4.)*(end-start);
45
46 double vx=end.x()-start.x();
47 double vy=end.y()-start.y();
48
49 double length = sqrt(vx*vx+vy*vy);
50 if (length==0) return;
51
52 // perpendicular vector
53 double px=-vy/length;
54 double py=vx/length;
55
56 // Arrow head lines go from end point
57 // to points about 3/4 of the total arrow, extending sideways about 1/4
58 // of the arrow length.
59
60
61 QPointF arwp1 = mid + QPointF( (int)( (length/4.)*px ),
62 (int)( (length/4.)*py ) );
63 QPointF arwp2 = mid - QPointF( (int)( (length/4.)*px ),
64 (int)( (length/4.)*py ) );
65
66 painter->setPen(pen());
67 //Draw arrow head
68 painter->drawLine( end, arwp1 );
69 painter->drawLine( end, arwp2 );
70 //Draw arrow line
71 painter->drawLine( start, end);
72
73 }
74};
